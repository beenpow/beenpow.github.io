### Point
- n이 주어집니다.
- 2 <= a, b, c 와 a * b * c = n 을 만족하는 서로 다른 수 a, b, c 를 출력합니다.

### Design
- 약수를 2번 구해줍니다.

### Complexity
- O(N)

### Code

```cpp
#include<iostream>
#include<vector>
#include<queue>
#include<map>
#include<algorithm>
#define rep(i, a, b) for(int i=(a); i<(b); i++)
#define r_rep(i, a, b) for(int i=(a); i >(b); i--)
typedef long long ll;
using namespace std;

ll n;
void solve(){
    cin >> n;
    for(ll i = 2; i * i <= n; i++){
        if(n%i == 0){
            // i and n/i
            for(ll j = 2; j * j <= i; j++){
                if(i % j == 0){
                    if(j == n/i || j == (i/j) || (i/j == n/i)) continue;
                    cout << "YES\n";
                    cout << j << " " << i/j << " " << n/i << '\n';
                    return;
                }
            }
            for(ll j = 2; j * j <= n/i; j++){
                if((n/i) % j == 0){
                    if(j == i || j == (n/i/j) || (i == (n/i/j))) continue;
                    cout << "YES\n";
                    cout << j << " " << (n/i/j) << " " << i << '\n';
                    return;
                }
            }
            
        }
    }
    cout << "NO\n";
}
int main(){
    //freopen("input.txt", "r", stdin);
    int tc; cin >> tc;
    while(tc--)
        solve();
    return 0;
}
```
