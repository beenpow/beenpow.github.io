#include<bits/stdc++.h>
#define fi first
#define se second
#define sz(x) (int)x.size()
#define all(x) x.begin(), x.end()
#define rep(i, a, b) for(int i = (a); i <(b); i++)
#define r_rep(i, a, b) for(int i = (a); i >(b); i--)
typedef long long ll;
using namespace std;

int solve() {
    int n, x; cin >> n >> x;
    vector<int> a(n + 1, 0);
    vector<int> _unsor;
    rep(i, 1, n + 1) {
        cin >> a[i];
        if (a[i-1] > a[i]) _unsor.push_back(i-1);
    }
    if (is_sorted(all(a))) return 0;
    
    int ans = 0, st = 1;
    auto t = lower_bound(all(a), x);
    if (t == a.end()) return -1;
    
    rep(i, st, n) {
        if (a[i] > x) {
            auto it = lower_bound(all(_unsor), i);
            if (it == _unsor.end()) break;
            
            int nFId = *it;
            if (nFId == i) {swap(a[i], x); ans++; continue;}
            if (x < a[nFId] && a[nFId - 1] <= x) ;
            else {
                swap(a[i], x);
                ans++;
                if (n == 500 && ans == 4) {
                    cout << "i" << i << '\n';
                    return -1;
                }
            }
        }
    }
    
    
    
    return is_sorted(all(a)) ? ans : -1;
}
int main(){
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    int tc; cin >> tc; while(tc--) cout << solve() << '\n';
    return 0;
}
